export const MotionLight = ({ intensity = 1, transition = { duration: 0.2 }, ...props }) => {
    const intensityMV = motion.useMotionValue(intensity);

    useEffect(() => {
        motion.animate(intensityMV, intensity, transition);
    }, [intensity]);

    class LightScript extends pc.Script {
        update() {
            this.entity.light.intensity = intensityMV.get();
        }
    }

    return <>
        <Script script={LightScript} />
        <Light {...props}/>
    </>
}

export const MouseRotation = () => {
    class MouseRotatesEntity extends pc.Script {
        initialize() {
            this.t = new pc.Vec2()
            this.c = new pc.Vec2()
            this.app.mouse.on(pc.EVENT_MOUSEMOVE, e => {
                this.t.set(e.x, e.y).mulScalar(2).subScalar(1).divScalar(20)
            });
        }

        update(dt){
            this.c.lerp(this.c, this.t, 0.4 * dt)
            this.entity.setEulerAngles(this.c.y, this.c.x, 0 )
        }
    }
    return <Script script={MouseRotatesEntity} />
}
