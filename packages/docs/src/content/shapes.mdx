export const ShapesPage = () => {

    // Define a color
    const shapes = [
        "box",
        "capsule",
        "cone",
        "cylinder",
        "plane",
        "sphere",
    ]
    
    const angularStep = Math.PI * 2 / shapes.length;
    const radius = 5;

    return (<Entity>

        {/* Create a camera */}
        <Entity name='camera'  >
            <Camera clearColor='red' />
            <OrbitControls />
        </Entity>

        {
            shapes.map((shape, i) => {

                const pos = [
                    Math.sin(i * angularStep) * radius,
                    Math.cos(i * angularStep) * radius,
                    0
                ]
            
                return <Entity key={shape} position={pos} scale={[3, 3, 3]}>
                    <Render type={shape} />
                </Entity>
            })
        }
    </Entity>)
}

<ShapesPage />
